get_driver <- function(Phi, quest, drv, top) {tibble(numero = rownames(Phi), carga = Phi[, drv]) %>%
        left_join(quest) %>%
        mutate(contribuicao = carga^2 / sum(carga^2)) %>%
        arrange(desc(contribuicao)) %>%
        head(n = top)}
